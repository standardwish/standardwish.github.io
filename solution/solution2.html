let total_price = 0;
 
function appendProduct(fruit, price) {
    total_price += price;
    const temp = document.createElement("div");
    temp.innerHTML = `<div>
                <li>${fruit}</li>
                </div>`;
    document.querySelector(".coloum").append(temp);
    document.querySelector(".price").textContent = `총 가격 : ${total_price}원`;
}
