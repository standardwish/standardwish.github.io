const 카운트 = document.getElementById("카운트");
let 숫자 = 0;
function 더하기() {
    숫자++;
    카운트.innerText = 숫자;
}
function 빼기() {
    숫자--;
    if (숫자 < 0) {
        숫자++;
        alert("음수가 될 수 없습니다.");
    }
    else {
        카운트.innerText = 숫자;
    }
}
